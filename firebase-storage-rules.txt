// Firebase Storage 보안 규칙 예시
// Firebase Console > Storage > Rules에서 설정해야 합니다

rules_version = '2';
service firebase.storage {
  match /b/{bucket}/o {
    // 갤러리 폴더에 대한 규칙
    match /gallery/{scheduleId}/{allPaths=**} {
      // 인증된 사용자는 읽기/쓰기 가능
      allow read, write: if request.auth != null;
      
      // 또는 더 구체적인 규칙 (선택사항)
      // allow read, write: if request.auth != null 
      //   && request.resource.size < 20 * 1024 * 1024  // 20MB 제한
      //   && request.resource.contentType.matches('image/.*');  // 이미지 파일만
    }
    
    // 기본 규칙 - 인증된 사용자만 접근
    match /{allPaths=**} {
      allow read, write: if request.auth != null;
    }
  }
}
